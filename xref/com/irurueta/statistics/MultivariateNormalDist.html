187     /**
188      * Sets covariance of this multivariate Gaussian distribution.
189      *
190      * @param cov                               covariance of this multivariate Gaussian distribution.
191      * @param validateSymmetricPositiveDefinite true if matrix must be
192      *                                          validated to be positive definite, false to skip validation.
193      * @throws InvalidCovarianceMatrixException if provided matrix is not
194      *                                          valid (nor square or symmetric positive definite if validation is
195      *                                          enabled).
196      */
197     public void setCovariance(final Matrix cov,
198                               final boolean validateSymmetricPositiveDefinite)
199             throws InvalidCovarianceMatrixException {
200         if (cov.getRows() != cov.getColumns()) {
201             throw new InvalidCovarianceMatrixException(
202                     "covariance matrix must be square");
203         }
204 
205         try {
206             if (validateSymmetricPositiveDefinite) {
207                 final CholeskyDecomposer decomposer = new CholeskyDecomposer(cov);
208                 decomposer.decompose();
209                 if (!decomposer.isSPD()) {
210                     throw new InvalidCovarianceMatrixException(
211                             "covariance matrix must be symmetric positive " +
212                                     "definite (non singular)");
213                 }
214             }
215 
216             mCov = new Matrix(cov);
217             mCovBasis = null;
218             mVariances = null;
219         } catch (final AlgebraException e) {
220             throw new InvalidCovarianceMatrixException(
221                     "covariance matrix must be square", e);
222         }
223     }

272     /**
273      * Indicates whether provided matrix is a valid covariance matrix.
274      * A valid covariance matrix must be square, symmetric and positive definite
275      * (i.e. non-singular).
276      *
277      * @param cov matrix to be checked.
278      * @return true if matrix is a valid covariance matrix, false otherwise.
279      */
280     public static boolean isValidCovariance(final Matrix cov) {
281         if (cov.getRows() != cov.getColumns()) {
282             return false;
283         }
284 
285         try {
286             final CholeskyDecomposer decomposer = new CholeskyDecomposer(cov);
287             decomposer.decompose();
288             return decomposer.isSPD();
289         } catch (final AlgebraException e) {
290             return false;
291         }
292     }

329     /**
330      * Evaluates the probability density function (p.d.f.) of a multivariate
331      * Gaussian distribution having current mean and covariance at point x.
332      *
333      * @param x array containing coordinates where p.d.f. is evaluated.
334      * @return evaluation of p.d.f.
335      * @throws NotReadyException            if this instance is not ready (mean and
336      *                                      covariance have not been provided or are not valid).
337      * @throws IllegalArgumentException     if provided point length is not valid.
338      * @throws DecomposerException          happens if covariance is numerically
339      *                                      unstable (i.e. contains NaNs or very large numbers).
340      * @throws RankDeficientMatrixException happens if covariance is singular.
341      */
342     public double p(final double[] x) throws NotReadyException,
343             DecomposerException, RankDeficientMatrixException {
344         if (!isReady()) {
345             throw new NotReadyException();
346         }
347 
348         final int k = x.length;
349         if (k != mMu.length) {
350             throw new IllegalArgumentException(
351                     "length of point must be equal to the length of mean");
352         }
353 
354         double detCov = 0.0;
355         try {
356             detCov = Utils.det(mCov);
357         } catch (final WrongSizeException ignore) {
358             // never thrown
359         }
360 
361         final double factor = 1.0 / (Math.sqrt(Math.pow(2.0 * Math.PI, k) *
362                 detCov));
363         return factor * Math.exp(-0.5 * squaredMahalanobisDistance(x));
364     }

366     /**
367      * Evaluates the cumulative distribution function (c.d.f.) of a Gaussian
368      * distribution having current mean and covariance values.
369      * The c.d.f. is equivalent to the joint probability of the multivariate
370      * Gaussian distribution of having a value less than x on each direction
371      * of the basis of independent variances obtained from covariance matrix.
372      * Because the c.d.f is a probability, it always returns values between 0.0
373      * and 1.0.
374      * NOTE: this method will resize provided basis instance if needed.
375      *
376      * @param x     point where c.d.f. is evaluated.
377      * @param basis instance where is stored the basis of each direction of
378      *              independent covariances, if provided.
379      * @return evaluation of c.d.f.
380      * @throws IllegalArgumentException if length of provided point is not equal
381      *                                  to length of current mean.
382      * @throws NotReadyException        if this instance is not ready (mean and
383      *                                  covariance have not been provided or are not valid).
384      * @throws DecomposerException      if covariance is numerically unstable (i.e.
385      *                                  contains NaNs or very large numbers).
386      */
387     public double cdf(final double[] x, final Matrix basis) throws NotReadyException,
388             DecomposerException {
389         if (!isReady()) {
390             throw new NotReadyException();
391         }
392 
393         final int k = x.length;
394         if (k != mMu.length) {
395             throw new IllegalArgumentException(
396                     "length of point must be equal to the length of mean");
397         }
398 
399         double p = 1.0;
400         try {
401             processCovariance();
402 
403             if (basis != null) {
404                 basis.copyFrom(mCovBasis);
405             }
406 
407             for (int i = 0; i < k; i++) {
408                 final double[] singleBasis =
409                         mCovBasis.getSubmatrixAsArray(0, i, k - 1, i);
410                 final double coordX = ArrayUtils.dotProduct(x, singleBasis);
411                 final double coordMu = ArrayUtils.dotProduct(mMu, singleBasis);
412                 p *= NormalDist.cdf(coordX, coordMu, Math.sqrt(mVariances[i]));
413             }
414 
415         } catch (final DecomposerException e) {
416             throw e;
417         } catch (final AlgebraException ignore) {
418             // never thrown
419         }
420 
421         return p;
422     }

446     /**
447      * Computes the joint probability of all probabilities provided in the
448      * array. The joint probability is computed by multiplying all components of
449      * the array, assuming that all probabilities are independent.
450      *
451      * @param p array containing probabilities for each independent variance
452      *          direction that can be obtained from provided covariance matrix.
453      * @return joint probability.
454      */
455     public static double jointProbability(final double[] p) {
456         double jointP = 1.0;
457         for (final double aP : p) {
458             jointP *= aP;
459         }
460         return jointP;
461     }

463     /**
464      * Evaluates the inverse cumulative distribution function of a multivariate
465      * Gaussian distribution for current mean and covariance values and provided
466      * probability values for each dimension of the multivariate Gaussian
467      * distribution.
468      * NOTE: this method will resize provided basis instance if needed.
469      *
470      * @param p      array containing probability values to evaluate the inverse
471      *               c.d.f. on each dimension. Values in the array must be between 0.0 and
472      *               1.0.
473      * @param result coordinates of the value x for which the c.d.f. has values
474      *               p.
475      * @param basis  instance where is stored the basis of each direction of
476      *               independent covariances, if provided.
477      * @throws IllegalArgumentException if length of probabilities is not equal
478      *                                  to mean length, or if result and length of probabilities are not equal,
479      *                                  or if provided probabilities are not between 0.0 and 1.0.
480      * @throws NotReadyException        if this instance is not ready (mean and
481      *                                  covariance have not been provided or are not valid).
482      * @throws DecomposerException      if covariance is numerically unstable (i.e.
483      *                                  contains NaNs or very large numbers).
484      */
485     public void invcdf(final double[] p, final double[] result, final Matrix basis)
486             throws NotReadyException, DecomposerException {
487         if (!isReady()) {
488             throw new NotReadyException(
489                     "mean and covariance not provided or invalid");
490         }
491 
492         final int k = p.length;
493         if (k != mMu.length) {
494             throw new IllegalArgumentException(
495                     "length of probabilities must be equal to the length of "
496                             + "mean");
497         }
498         if (k != result.length) {
499             throw new IllegalArgumentException("length of result must be equal "
500                     + "to the length of mean");
501         }
502 
503         try {
504             processCovariance();
505 
506             if (basis != null) {
507                 basis.copyFrom(mCovBasis);
508             }
509 
510             // initialize to mean
511             System.arraycopy(mMu, 0, result, 0, k);
512             for (int i = 0; i < k; i++) {
513                 final double[] singleBasis =
514                         mCovBasis.getSubmatrixAsArray(0, i, k - 1, i);
515                 final double coord = NormalDist.invcdf(p[i], mMu[i],
516                         Math.sqrt(mVariances[i])) - mMu[i];
517                 // coord*singleBasis
518                 ArrayUtils.multiplyByScalar(singleBasis, coord, singleBasis);
519 
520                 // result = mean + coord*singleBasis
521                 ArrayUtils.sum(result, singleBasis, result);
522             }
523         } catch (final DecomposerException e) {
524             throw e;
525         } catch (final AlgebraException ignore) {
526             // never thrown
527         }
528     }

609     /**
610      * Evaluates the inverse cumulative distribution function of a multivariate
611      * Gaussian distribution for current mean and covariance values and provided
612      * probability value.
613      * Obtained result coordinates are computed taking into account the basis
614      * of independent variances computed from current covariance matrix.
615      * NOTE: notice that the inverse cdf of a mutivariate Gaussian distribution
616      * does not have a unique solution. This method simply returns one of the
617      * possible solutions by assuming equal probabilities on each dimension.
618      * NOTE: this method will resize provided basis instance if needed.
619      *
620      * @param p      probability value to evaluate the inverse c.d.f. at. This value
621      *               must be between 0.0 and 1.0
622      * @param result coordinates of the value x for which the c.d.f. has value
623      *               p.
624      * @param basis  instance where is stored the basis of each direction of
625      *               independent covariances, if provided.
626      * @throws IllegalArgumentException if provided probability value is not
627      *                                  between 0.0 and 1.0, if length of provided result array is not equal
628      *                                  to length of current mean.
629      * @throws NotReadyException        if this instance is not ready (mean and
630      *                                  covariance have not been provided or are not valid).
631      * @throws DecomposerException      if covariance is numerically unstable (i.e.
632      *                                  contains NaNs or very large numbers).
633      */
634     public void invcdf(final double p, final double[] result, final Matrix basis)
635             throws NotReadyException, DecomposerException {
636         if (p <= 0.0 || p >= 1.0) {
637             throw new IllegalArgumentException(
638                     "probability value must be between 0.0 and 1.0");
639         }
640 
641         if (!isReady()) {
642             throw new NotReadyException(
643                     "mean and covariance not provided or invalid");
644         }
645 
646         final int k = result.length;
647         if (k != mMu.length) {
648             throw new IllegalArgumentException(
649                     "length of result must be equal to mean length");
650         }
651 
652         final double[] probs = new double[k];
653         Arrays.fill(probs, Math.pow(p, 1.0 / k));
654         invcdf(probs, result, basis);
655     }

657     /**
658      * Evaluates the inverse cumulative distribution function of a multivariate
659      * Gaussian distribution for current mean and covariance values and provided
660      * probability value.
661      * Obtained result coordinates are computed taking into account the basis
662      * of independent variances computed from current covariance matrix.
663      * NOTE: notice that the inverse cdf of a mutivariate Gaussian distribution
664      * does not have a unique solution. This method simply returns one of the
665      * possible solutions by assuming equal probabilities on each dimension.
666      * NOTE: this method will resize provided basis instance if needed.
667      *
668      * @param p     probability value to evaluate the inverse c.d.f. at. This value
669      *              must be between 0.0 and 1.0
670      * @param basis instance where is stored the basis of each direction of
671      *              independent covariances, if provided.
672      * @return a new array containing the coordinates of the value x for which
673      * the c.d.f. has value p.
674      * @throws IllegalArgumentException if provided probability value is not
675      *                                  between 0.0 and 1.0.

759     /**
760      * Computes the squared Mahalanobis distance of provided multivariate pot x
761      * for current mean and covariance values.
762      *
763      * @param x point where Mahalanobis distance is evaluated.
764      * @return Mahalanobis distance of provided point respect to mean.
765      * @throws DecomposerException          happens if covariance is numerically
766      *                                      unstable (i.e. contains NaNs or very large numbers).
767      * @throws RankDeficientMatrixException happens if covariance is singular.
768      */
769     public double squaredMahalanobisDistance(final double[] x) throws DecomposerException,
770             RankDeficientMatrixException {
771         final double[] diff = ArrayUtils.subtractAndReturnNew(x, mMu);
772         final Matrix diffMatrix = Matrix.newFromArray(diff, true);
773         final Matrix transDiffMatrix = diffMatrix.transposeAndReturnNew();
774 
775         try {
776             final Matrix invCov = Utils.inverse(mCov);
777             transDiffMatrix.multiply(invCov);
778             transDiffMatrix.multiply(diffMatrix);
779 
780         } catch (final WrongSizeException ignore) {
781             // never thrown
782         }
783 
784         return transDiffMatrix.getElementAtIndex(0);
785     }

787     /**
788      * Processes current covariance by decomposing it into a basis and its
789      * corresponding variances if needed.
790      *
791      * @throws DecomposerException   happens if covariance is numerically
792      *                               unstable (i.e. contains NaNs or very large numbers).
793      * @throws NotReadyException     never thrown because decomposer will always be
794      *                               ready.
795      * @throws LockedException       never thrown because decomposer never will be
796      *                               locked.
797      * @throws NotAvailableException never thrown because first a
798      *                               DecomposerException will be thrown before attempting to get V or
799      *                               singular values.
800      */
801     public void processCovariance() throws DecomposerException,
802             NotReadyException, LockedException, NotAvailableException {
803         if (mCov == null) {
804             throw new NotReadyException("covariance must be defined");
805         }
806 
807         if (mCovBasis == null || mVariances == null) {
808             final SingularValueDecomposer decomposer =
809                     new SingularValueDecomposer(mCov);
810             decomposer.decompose();
811 
812             // because matrix is symmetric positive definite:
813             // And matrices U and V are orthonormal
814             // Cov = A'*A = (U*S*V')'*(U*S*V')=V*S*U'*U*S*V' = V*S^2*V',
815 
816             // where matrix S is diagonal, and contains the standard deviations
817             // on each direction of the basis V, and hence S^2 is also diagonal but
818             // containing variances on each direction.
819             // The values of S^2 are the eigenvalues of Cov, and V are the
820             // eigenvectors of Cov, hence covariance can be expressed as variances
821             // on each direction of the basis V.
822 
823             // matrix containing eigenvectors (basis of directions)
824             mCovBasis = decomposer.getV();
825 
826             // array containing the eigenvalues (variances on each direction)
827             mVariances = decomposer.getSingularValues();
828         }
829     }

831     /**
832      * Evaluates the Jacobian and a multivariate function at a certain mean
833      * point and computes the non-linear propagation of Gaussian uncertainty
834      * through such function at such point.
835      *
836      * @param evaluator  interface to evaluate a multivariate function and its
837      *                   Jacobian at a certain point.
838      * @param mean       mean of original multivariate Gaussian distribution to be
839      *                   propagated. Must have the length of the number of input variables of the
840      *                   multivariate function to be evaluated.
841      * @param covariance covariance of original Gaussian distribution to be
842      *                   propagated. Must be symmetric positive definite having size NxN where N
843      *                   is the length of provided mean.
844      * @param result     instance where propagated multiavariate Gaussian
845      *                   distribution will be stored.
846      * @throws WrongSizeException               if evaluator returns an invalid number of
847      *                                          variables (i.e. negative or zero).
848      * @throws InvalidCovarianceMatrixException if provided covariance matrix is
849      *                                          not valid (i.e. is not symmetric positive definite).
850      * @see <a href="https://github.com/joansola/slamtb">propagateUncertainty.m at https://github.com/joansola/slamtb</a>
851      */
852     public static void propagate(final JacobianEvaluator evaluator, final double[] mean,
853                                  final Matrix covariance, final MultivariateNormalDist result)
854             throws WrongSizeException, InvalidCovarianceMatrixException {
855 
856         final int ndims = mean.length;
857         final int nvars = evaluator.getNumberOfVariables();
858         final double[] evaluation = new double[nvars];
859         final Matrix jacobian = new Matrix(nvars, ndims);
860         evaluator.evaluate(mean, evaluation, jacobian);
861 
862         // [y, Y_x] = f(x)
863         // Y = Y_x * X * Y_x'
864         final Matrix jacobianTrans = jacobian.transposeAndReturnNew();
865         jacobian.multiply(covariance);
866         jacobian.multiply(jacobianTrans);
867 
868         // ensure that new covariance is symmetric positive definite
869         jacobian.symmetrize();
870 
871         result.setMean(evaluation);
872         result.setCovariance(jacobian, false);
873     }

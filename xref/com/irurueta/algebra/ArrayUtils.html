211     /**
212      * Computes the dot product of provided arrays as the sum of the product
213      * of the elements of both arrays.
214      *
215      * @param firstOperand  First operand.
216      * @param secondOperand Second operand.
217      * @return Dot product.
218      * @throws IllegalArgumentException Raised if first and second operands
219      *                                  arrays don't have the same length.
220      */
221     public static double dotProduct(
222             final double[] firstOperand, final double[] secondOperand) {
223         if (firstOperand.length != secondOperand.length) {
224             throw new IllegalArgumentException(
225                     "both operands must have same length");
226         }
227 
228         double result = 0.0;
229         for (int i = 0; i < firstOperand.length; i++) {
230             result += firstOperand[i] * secondOperand[i];
231         }
232         return result;
233     }

235     /**
236      * Computes the dot product of provided arrays as the sum of the product of
237      * the elements of both arrays.
238      *
239      * @param firstOperand   first operand.
240      * @param secondOperand  second operand.
241      * @param jacobianFirst  matrix where jacobian of first operand will be
242      *                       stored. Must be a column matrix having the same number of rows as the
243      *                       first operand length.
244      * @param jacobianSecond matrix where jacobian of second operand will be
245      *                       stored. Must be a column matrix having the same number of rows as the
246      *                       second operand length.
247      * @return dot product.
248      * @throws IllegalArgumentException if first and second operands don't have
249      *                                  the same length or if jacobian matrices are not column vectors having
250      *                                  the same length as their respective operands.
251      */
252     public static double dotProduct(
253             final double[] firstOperand, final double[] secondOperand,
254             final Matrix jacobianFirst, final Matrix jacobianSecond) {
255         if (jacobianFirst != null &&
256                 (jacobianFirst.getRows() != 1 ||
257                         jacobianFirst.getColumns() != firstOperand.length)) {
258             throw new IllegalArgumentException("jacobian first must be a " +
259                     "row vector having the same number of columns as " +
260                     "first operand length");
261         }
262         if (jacobianSecond != null &&
263                 (jacobianSecond.getRows() != 1 ||
264                         jacobianSecond.getColumns() != secondOperand.length)) {
265             throw new IllegalArgumentException("jacobian second must be a " +
266                     "row vector having the same number of columns as " +
267                     "second operand length");
268         }
269 
270         if (jacobianFirst != null) {
271             jacobianFirst.setSubmatrix(0, 0, 0, firstOperand.length - 1,
272                     firstOperand);
273         }
274         if (jacobianSecond != null) {
275             jacobianSecond.setSubmatrix(0, 0, 0, secondOperand.length - 1,
276                     secondOperand);
277         }
278 
279         return dotProduct(firstOperand, secondOperand);
280     }

282     /**
283      * Computes the angle between two vectors.
284      * The angle is defined between 0 and PI.
285      *
286      * @param firstOperand  first operand.
287      * @param secondOperand second operand.
288      * @return angle between arrays.
289      * @throws IllegalArgumentException if first and second operands don't have
290      *                                  the same length.
291      */
292     public static double angle(final double[] firstOperand, final double[] secondOperand) {
293         final double norm1 = Utils.normF(firstOperand);
294         final double norm2 = Utils.normF(secondOperand);
295         return Math.acos(Math.min(dotProduct(firstOperand, secondOperand) / norm1 / norm2, 1.0));
296     }

519     /**
520      * Normalizes provided array and computes corresponding jacobian.
521      *
522      * @param v        array to be normalized.
523      * @param result   array where result of normalized array will be stored.
524      * @param jacobian matrix where jacobian will be stored.
525      * @throws IllegalArgumentException if provided arrays don't have the same
526      *                                  length or if provided jacobian is not NxN, where N is length of arrays.
527      */
528     public static void normalize(final double[] v, final double[] result, final Matrix jacobian) {
529         final double s = v.length;
530 
531         if (s != result.length) {
532             throw new IllegalArgumentException(
533                     "both arrays must have the same length");
534         }
535 
536         if (jacobian != null && (jacobian.getRows() != s ||
537                 jacobian.getColumns() != s)) {
538             throw new IllegalArgumentException(
539                     "provided jacobian is not NxN where N is length of v");
540         }
541 
542         final double n2 = dotProduct(v, v);
543         final double n = Math.sqrt(n2);
544 
545         if (jacobian != null) {
546             try {
547                 final double n3 = n * n2;
548 
549                 // VN_v = (n2*eye(s) - v*v') / n3
550                 Matrix.identity(jacobian);
551                 jacobian.multiplyByScalar(n2);
552                 jacobian.subtract(Matrix.newFromArray(v, true).
553                         multiplyAndReturnNew(Matrix.newFromArray(v, false)));
554                 if (n3 != 0.0) {
555                     jacobian.multiplyByScalar(1.0 / n3);
556                 } else {
557                     jacobian.initialize(Double.MAX_VALUE);
558                 }
559             } catch (final WrongSizeException ignore) {
560                 // never thrown
561             }
562         }
563 
564         if (n != 0.0) {
565             internalMultiplyByScalar(v, 1.0 / n, result);
566         } else {
567             Arrays.fill(result, Double.MAX_VALUE);
568         }
569     }

630     /**
631      * Reverses the order of elements in the array.
632      *
633      * @param v      array to be reversed.
634      * @param result instance where results will be stored.
635      * @throws IllegalArgumentException if provided arrays don't have the same
636      *                                  length.
637      */
638     public static void reverse(final double[] v, final double[] result) {
639 
640         final int length = v.length;
641 
642         if (result.length != length) {
643             throw new IllegalArgumentException();
644         }
645 
646         final int halfLength = length / 2;
647         double tmp;
648         for (int i = 0, j = length - 1; i < halfLength; i++, j--) {
649             tmp = v[i];
650             result[i] = v[j];
651             result[j] = tmp;
652         }
653 
654         if (length % 2 != 0) {
655             // if length is odd, copy central value
656             result[halfLength] = v[halfLength];
657         }
658     }

695         if (result.length != length) {
696             throw new IllegalArgumentException();
697         }
698 
699         final int halfLength = length / 2;
700         Complex tmp;
701         if (v != result) {
702             // for different instances, copy values
703             for (int i = 0, j = length - 1; i < halfLength; i++, j--) {
704                 tmp = new Complex(v[i]);
705                 result[i] = new Complex(v[j]);
706                 result[j] = tmp;
707             }
708 
709             if (length % 2 != 0) {
710                 // if length is odd, copy central value
711                 result[halfLength] = new Complex(v[halfLength]);
712             }
713         } else {
714             // for same instances, rearrange values
715             for (int i = 0, j = length - 1; i < halfLength; i++, j--) {
716                 tmp = v[i];
717                 result[i] = v[j];
718                 result[j] = tmp;
719             }
720 
721             if (length % 2 != 0) {
722                 // if length is odd, copy central value
723                 result[halfLength] = v[halfLength];
724             }
725         }
726     }

872      * @param result array of length 2 containing found minimum and maximum
873      *               values at positions 0 and 1 respectively.
874      * @param pos    array of length 2 containing positions where minimum and
875      *               maximum where found in v array. Position 0 will contain minimum position,
876      *               position 1 will contain maximum position.
877      * @throws IllegalArgumentException if provided result or pos array don't
878      *                                  have length 2.
879      */
880     public static void minMax(final double[] v, final double[] result, final int[] pos) {
881         if (result.length != 2) {
882             throw new IllegalArgumentException("result must have length 2");
883         }
884         if (pos != null && pos.length != 2) {
885             throw new IllegalArgumentException("pos must have length 2");
886         }
887 
888         final int length = v.length;
889         double min = Double.MAX_VALUE;
890         double max = -Double.MAX_VALUE;
891         int minPos = -1;
892         int maxPos = -1;
893         for (int i = 0; i < length; i++) {
894             if (v[i] < min) {
895                 min = v[i];
896                 minPos = i;
897             }
898             if (v[i] > max) {
899                 max = v[i];
900                 maxPos = i;
901             }
902         }
903 
904         result[0] = min;
905         result[1] = max;
906         if (pos != null) {
907             pos[0] = minPos;
908             pos[1] = maxPos;
909         }

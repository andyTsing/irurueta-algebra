16  package com.irurueta.algebra;

18  import com.irurueta.statistics.UniformRandomizer;
19  import org.junit.Test;
20  
21  import java.util.Arrays;
22  import java.util.Random;
23  
24  import static org.junit.Assert.*;
25  
26  public class GaussJordanEliminationTest {
27  
28      public static final double MIN_RANDOM_VALUE = 0.0;
29      public static final double MAX_RANDOM_VALUE = 50.0;
30  
31      public static final int MIN_COLUMNS = 1;
32      public static final int MAX_COLUMNS = 50;
33  
34      public static final double ABSOLUTE_ERROR = 1e-6;
35  
36      @Test
37      public void testProcessMatrix() throws WrongSizeException,
38              SingularMatrixException, RankDeficientMatrixException,
39              DecomposerException {
40          final UniformRandomizer randomizer = new UniformRandomizer(new Random());
41          final int rows = randomizer.nextInt(MIN_COLUMNS + 5, MAX_COLUMNS + 5);
42          final int colsB = randomizer.nextInt(MIN_COLUMNS, MAX_COLUMNS);
43  
44          Matrix a;
45          Matrix b;
46          final Matrix a2;
47          final Matrix b2;
48          final Matrix invA;
49          final Matrix x;
50  
51          // Test for non-singular square matrix
52          a = DecomposerHelper.getNonSingularMatrixInstance(rows, rows);
53          b = Matrix.createWithUniformRandomValues(rows, colsB, MIN_RANDOM_VALUE,
54                  MAX_RANDOM_VALUE);
55  
56          a2 = new Matrix(a);
57          b2 = new Matrix(b);
58  
59          GaussJordanElimination.process(a2, b2);
60  
61          // check correctness
62          invA = Utils.inverse(a);
63          x = Utils.solve(a, b);
64  
65          assertTrue(a2.equals(invA, ABSOLUTE_ERROR));
66          assertTrue(b2.equals(x, ABSOLUTE_ERROR));
67  
68          // Force WrongSizeException
69  
70          // non square matrix a
71          a = new Matrix(rows, rows + 1);
72          b = new Matrix(rows, colsB);
73          try {
74              GaussJordanElimination.process(a, b);
75              fail("WrongSizeException expected but not thrown");
76          } catch (final WrongSizeException ignore) {
77          }
78  
79          // different rows
80          a = new Matrix(rows, rows);
81          b = new Matrix(rows + 1, colsB);
82          try {
83              GaussJordanElimination.process(a, b);
84              fail("WrongSizeException expected but not thrown");
85          } catch (final WrongSizeException ignore) {
86          }
87  
88          // Force SingularMatrixException
89          a = new Matrix(rows, rows);
90          b = Matrix.createWithUniformRandomValues(rows, colsB, MIN_RANDOM_VALUE,
91                  MAX_RANDOM_VALUE);
92          try {
93              GaussJordanElimination.process(a, b);
94              fail("SingularMatrixException expected but not thrown");
95          } catch (final SingularMatrixException ignore) {
96          }
97      }
98  
99      @Test
100     public void testProcessArray() throws WrongSizeException,
101             SingularMatrixException, RankDeficientMatrixException,
102             DecomposerException {
103         final UniformRandomizer randomizer = new UniformRandomizer(new Random());
104         final int rows = randomizer.nextInt(MIN_COLUMNS + 5, MAX_COLUMNS + 5);
105 
106         Matrix a;
107         final Matrix a2;
108         final Matrix invA;
109         double[] b;
110         final double[] b2;
111         final double[] x;
112 
113         // Test for non-singular square matrix
114         a = DecomposerHelper.getNonSingularMatrixInstance(rows, rows);
115         b = new double[rows];
116         randomizer.fill(b, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
117 
118         a2 = new Matrix(a);
119         b2 = Arrays.copyOf(b, rows);
120 
121         GaussJordanElimination.process(a2, b2);
122 
123         // check correctness
124         invA = Utils.inverse(a);
125         x = Utils.solve(a, b);
126 
127         assertTrue(a2.equals(invA, ABSOLUTE_ERROR));
128         assertArrayEquals(b2, x, ABSOLUTE_ERROR);
129 
130         // Force WrongSizeException
131 
132         // non square matrix a
133         a = new Matrix(rows, rows + 1);
134         b = new double[rows];
135         try {
136             GaussJordanElimination.process(a, b);
137             fail("WrongSizeException expected but not thrown");
138         } catch (final WrongSizeException ignore) {
139         }
140 
141         // different lengths
142         a = new Matrix(rows, rows);
143         b = new double[rows + 1];
144         try {
145             GaussJordanElimination.process(a, b);
146             fail("WrongSizeException expected but not thrown");
147         } catch (final WrongSizeException ignore) {
148         }
149 
150         // Force SingularMatrixException
151         a = new Matrix(rows, rows);
152         b = new double[rows];
153         try {
154             GaussJordanElimination.process(a, b);
155             fail("SingularMatrixException expected but not thrown");
156         } catch (final SingularMatrixException ignore) {
157         }
158     }
